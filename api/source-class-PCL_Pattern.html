 50:     /**
 51:      * PCL_Pattern($patternID)
 52:      *
 53:      * Initializes PCL_Pattern and sets the patternID for the instance.
 54:      *
 55:      * @param   int $patternID pattern id
 56:      * @access  public
 57:      * @return  object
 58:      */
 59:     public function __construct($patternID)
 60:     {
 61:         $this->patternID = (string)$patternID;
 62:     }

 64:     /**
 65:      * get()
 66:      *
 67:      * Preforms a curl call. get() returns an array based on the XML data recieved from the call.
 68:      *
 69:      * @access  public
 70:      * @return  array
 71:      */
 72:     public function get()
 73:     {
 74:         $url = "http://www.colourlovers.com/api/pattern/" . $this->patternID;
 75:         if (function_exists("curl_init")) {
 76:             $ch = curl_init();
 77:             curl_setopt($ch, CURLOPT_URL, $url);
 78:             curl_setopt($ch, CURLOPT_HEADER, false);
 79:             curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
 80:             $xml = curl_exec($ch);
 81:             curl_close($ch);
 82:         } else {
 83:             $xml = file_get_contents($url);
 84:         }
 85:         return $this->parse($xml);
 86:     }
 87: 
 88:     /**
 89:      * parse($xml)
 90:      *
 91:      * Parses XML data into an array to work with the returned curl call. Fixes the problem
 92:      * with SimpleXML and CDATA and also HTML entities by base64 encoding CDATA values.
 93:      *
 94:      * @param   string $xml xml string
 95:      * @access  private
 96:      * @return  array
 97:      */
 98:     private function parse($xml)
 99:     {
100:         $xml = preg_replace("/\<\!\[CDATA\[(.*?)\]\]\>/ies", "'[CDATA]'.base64_encode('$1').'[/CDATA]'", $xml);
101:         $xml = new SimpleXMLElement($xml);
102:         if (isset($xml->noresults)) {
103:             return false;
104:         }
105:         $out = array();
106:         foreach ($xml->pattern as $pattern) {
107:             $p = array();
108:             $p['title'] = $this->parseString($pattern->title);
109:             $p['lover'] = $this->parseString($pattern->userName);
110:             $p['views'] = $this->parseInt($pattern->numViews);
111:             $p['votes'] = $this->parseInt($pattern->numVotes);
112:             $p['comments'] = $this->parseInt($pattern->numComments);
113:             $p['hearts'] = $this->parseInt($pattern->numHearts);
114:             $p['rank'] = $this->parseInt($pattern->rank);
115:             $p['date'] = $this->parseString($pattern->dateCreated);
116:             $p['colors'] = array();
117:             foreach ($pattern->colors->hex as $hex) {
118:                 $p['colors'][] = $this->parseString($hex);
119:             }
120:             $p['description'] = $this->parseString($pattern->description);
121:             $p['url'] = $this->parseString($pattern->url);
122:             $p['image'] = $this->parseString($pattern->imageUrl);
123:             $p['api'] = $this->parseString($pattern->apiUrl);
124:             $out[] = $p;
125:         }
126:         $out = $out[0];
127:         return $out;
128:     }
129: 
130:     /**
131:      * parseString($obj_val)
132:      *
133:      * Parses a value passed from SimpleXML into a string. If needed, CDATA will be base64 decoded.
134:      *
135:      * @param   object $obj_val SimpleXML value
136:      * @access  private
137:      * @return  string
138:      */
139:     private function parseString($obj_val)
140:     {
141:         $obj_val = preg_replace("/\[CDATA\](.*?)\[\/CDATA\]/ies", "base64_decode('$1')", $obj_val);
142:         return (string)stripslashes($obj_val);
143:     }
144: 
145:     /**
146:      * parseInt($obj_val)
147:      *
148:      * Parses a value passed from SimpleXML into a string.
149:      *
150:      * @param   object $obj_val SimpleXML value
151:      * @access  private
152:      * @return  string
153:      */
154:     private function parseInt($obj_val)
155:     {
156:         $obj_val = preg_replace("/[^0-9]/", "", $obj_val);
157:         return (int)$obj_val;
158:     }
