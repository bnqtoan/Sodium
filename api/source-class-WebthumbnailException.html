334:     /**
335:      * Create and execute a http call.
336:      * @param string $url
337:      * @throws WebthumbnailException
338:      */
339:     public function __construct($url)
340:     {
341:         if (!function_exists('curl_init')) {
342:             throw new WebthumbnailException("Curl is not supported by your version of PHP!");
343:         }
344: 
345:         $ch = curl_init($url);
346:         if (!$ch) {
347:             throw new WebthumbnailException("curl_init failed!");
348:         }
349: 
350:         $referer = '-';
351:         if (isset($_SERVER) && isset($_SERVER['HTTP_REFERER'])) {
352:             $referer = $_SERVER['HTTP_REFERER'];
353:         }
354: 
355:         $userAgent = 'Webthumbnail.org Client PHP/' . phpversion();
356:         if (isset($_SERVER) && isset($_SERVER['SERVER_SOFTWARE'])) {
357:             $userAgent .= ' ' . $_SERVER['SERVER_SOFTWARE'];
358:         }
359: 
360:         curl_setopt($ch, CURLOPT_REFERER, $referer);
361:         curl_setopt($ch, CURLOPT_USERAGENT, $userAgent);
362:         curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
363: 
364:         if (!ini_get('open_basedir') && !ini_get('safe_mode')) {
365:             curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
366:         }
367: 
368:         $this->_response = curl_exec($ch);
369:         if (!$this->_response) {
370:             throw new WebthumbnailException("curl_exec failed!");
371:         }
372: 
373:         $this->_statusCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
374:         $this->_contentType = curl_getinfo($ch, CURLINFO_CONTENT_TYPE);
375:         $this->_contentLength = curl_getinfo($ch, CURLINFO_CONTENT_LENGTH_DOWNLOAD);
376: 
377:         curl_close($ch);
378:     }
