 68:      * @access  public
 69:      * @return  object

 71:     public function __construct($lover)
 72:     {
 73:         $this->lover = str_replace(" ", "+", $lover);

 77:      * comments()

 79:      * When called, sets $comments to true.

 81:      * @access  public
 82:      * @return  void

 84:     public function comments()
 85:     {
 86:         $this->comments = true;
 87:     }
 88: 
 89:     /**
 90:      * get()
 91:      *
 92:      * Preforms a curl call. get() returns an array based on the XML data recieved from the call.
 93:      *
 94:      * @access  public
 95:      * @return  array
 96:      */
 97:     public function get()
 98:     {
 99:         $url = $this->url();
100:         if (function_exists("curl_init")) {
101:             $ch = curl_init();
102:             curl_setopt($ch, CURLOPT_URL, $url);
103:             curl_setopt($ch, CURLOPT_HEADER, false);
104:             curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
105:             $xml = curl_exec($ch);
106:             curl_close($ch);
107:         } else {
108:             $xml = file_get_contents($url);
109:         }
110:         return $this->parse($xml);
111:     }
112: 
113:     /**
114:      * url()
115:      *
116:      * Sets the API url and sets the comments parameter if $comments is TRUE.
117:      *
118:      * @access  private
119:      * @return  string
120:      */
121:     private function url()
122:     {
123:         $url = "http://www.colourlovers.com/api/lover/" . $this->lover;
124:         if ($this->comments) {
125:             $url .= "?comments=1";
126:         }
127:         return $url;
128:     }

130:     /**
131:      * parse($xml)
132:      *
133:      * Parses XML data into an array to work with the returned curl call. Fixes the problem
134:      * with SimpleXML and CDATA and also HTML entities by base64 encoding CDATA values.
135:      *
136:      * @param   string $xml xml string
137:      * @access  private
138:      * @return  array
139:      */
140:     private function parse($xml)
141:     {
142:         $xml = preg_replace("/\<\!\[CDATA\[(.*?)\]\]\>/ies", "'[CDATA]'.base64_encode('$1').'[/CDATA]'", $xml);
143:         $xml = new SimpleXMLElement($xml);
144:         if (isset($xml->noresults)) {
145:             return false;
146:         }
147:         $out = array();
148:         foreach ($xml->lover as $lover) {
149:             $l = array();
150:             $l['lover'] = $this->parseString($lover->userName);
151:             $l['registered'] = $this->parseString($lover->dateRegistered);
152:             $l['activeLast'] = $this->parseString($lover->dateLastActive);
153:             $l['rating'] = $this->parseInt($lover->rating);
154:             $l['location'] = $this->parseString($lover->location);
155:             $l['colors'] = $this->parseInt($lover->numColors);
156:             $l['palettes'] = $this->parseInt($lover->numPalettes);
157:             $l['patterns'] = $this->parseInt($lover->numPatterns);
158:             $l['commentsMade'] = $this->parseInt($lover->numCommentsOnProfile);
159:             $l['profileComments'] = $this->parseInt($lover->numCommentsOnProfile);
160:             $l['lovers'] = $this->parseInt($lover->numLovers);
161:             if ($this->comments) {
162:                 $l['comments'] = array();
163:                 foreach ($lover->comments->comment as $comment) {
164:                     $c = array();
165:                     $c['date'] = $this->parseString($comment->commentDate);
166:                     $c['lover'] = $this->parseString($comment->commentUserName);
167:                     $c['comment'] = $this->parseString($comment->commentComments);
168:                     $l['comments'][] = $c;
169:                 }
170:             }
171:             $l['url'] = $this->parseString(str_replace(" ", "+", $lover->url));
172:             $l['api'] = $this->parseString(str_replace(" ", "+", $lover->apiUrl));
173:             $out[] = $l;
174:         }
175:         $out = $out[0];
176:         return $out;
177:     }
178: 
179:     /**
180:      * parseString($obj_val)
181:      *
182:      * Parses a value passed from SimpleXML into a string. If needed, CDATA will be base64 decoded.
183:      *
184:      * @param   object $obj_val SimpleXML value
185:      * @access  private
186:      * @return  string
187:      */
188:     private function parseString($obj_val)
189:     {
190:         $obj_val = preg_replace("/\[CDATA\](.*?)\[\/CDATA\]/ies", "base64_decode('$1')", $obj_val);
191:         return (string)stripslashes($obj_val);
192:     }
193:

197:      * Parses a value passed from SimpleXML into a string.

202:      */
203:     private function parseInt($obj_val)
204:     {
205:         $obj_val = preg_replace("/[^0-9]/", "", $obj_val);
206:         return (int)$obj_val;
207:     }
