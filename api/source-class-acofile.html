 38:     {
 39:         if (!is_array($param)) {
 40:             $this->i++;

 42:             $this->colors[$this->i]['g'] = $g;

 52:     {
 53:         return sprintf("%c%c", ($x >> 8) & 0xff, $x & 0xff);

 56:     function createAcofile()
 57:     {
 58:         $this->count = count($this->colors);
 59:         $out = $this->n(1);
 60:         $out .= $this->n($this->count);
 61:         for ($k = 1; $k <= $this->i; $k++) {
 62:             $out .= $this->n(0);
 63:             $out .= $this->n(($this->colors[$k]['r'] << 8) | $this->colors[$k]['r']);
 64:             $out .= $this->n(($this->colors[$k]['g'] << 8) | $this->colors[$k]['g']);
 65:             $out .= $this->n(($this->colors[$k]['b'] << 8) | $this->colors[$k]['b']);
 66:             $out .= $this->n(0);
 67:         }

 70:         for ($l = 1; $l <= $this->i; $l++) {
