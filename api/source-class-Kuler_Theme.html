 30: /**
 31:  * Provides encapsulation for an XML Theme retrieved from the API.
 32:  *
 33:  * @package PHP-Kuler-API-Library
 34:  */
 35: class Kuler_Theme
 36: {
 37: 
 38:     public $_theme;
 39: 
 40:     /**
 41:      * Enables creation of a Kuler_Theme object out of a theme XML Item.
 42:      * Checks the existance of the 'themeID' field to make sure the item was
 43:      * loaded correctly.
 44:      *
 45:      * @param SimpleXMLElement $xmlItem
 46:      */
 47:     public function __construct(SimpleXMLElement $rssItem)
 48:     {
 49:         $namespaces = $rssItem->getDocNamespaces();
 50:         $this->_theme = $rssItem->children($namespaces['kuler'])->themeItem;
 51:         if (!isset($this->_theme->themeID))
 52:             throw new Kuler_Exception('Error loading the Kuler item');
 53:     }
 54: 
 55:     /**
 56:      * Automatic accessors for the $_theme object. The return value is a
 57:      * SimpleXMLElement and should be properly casted in most situations.
 58:      *
 59:      * @param string $name
 60:      * @return SimpleXMLElement The value of retrieving the "theme{$name}" property from the internal $_theme object.
 61:      */
 62:     public function __get($name)
 63:     {
 64:         $property = 'theme' . ucfirst($name);
 65:         if (property_exists($this->_theme, $property))
 66:             return $this->_theme->$property;
 67:         else
 68:             throw new Kuler_Exception('Property \'' . $name . '\' not found');
 69:     }
 70: 
 71:     /**
 72:      * Retrieves the theme's author's name
 73:      */
 74:     public function getAuthorLabel()
 75:     {
 76:         return (string)$this->_theme->themeAuthor->authorLabel;
 77:     }
 78: 
 79:     /**
 80:      * Retrieves the theme's author's Kuler ID
 81:      */
 82:     public function getAuthorID()
 83:     {
 84:         return (int)$this->_theme->themeAuthor->authorID;
 85:     }
 86: 
 87:     /**
 88:      * Retrieves the theme's title
 89:      */
 90:     public function getTitle()
 91:     {
 92:         return (string)$this->_theme->themeTitle;
 93:     }
 94: 
 95:     /**
 96:      * Returns an Array of strings where each string is the HEX color of a swatch
 97:      * in the theme.
 98:      *
 99:      * @param boolean $prependPound If set, it will prepend the pound symbol (#)
100:      * to each color - useful for direct HTML usage for example.
101:      * @return Array Each swatch in the theme in its hexadecimal format.
102:      */
103:     public function getSwatchesHex($prependPound = true)
104:     {
105:         $hex = array();
106:         foreach ($this->swatches->swatch as $swatch) {
107:             $hex[] = ($prependPound ? '#' : '') . (string)$swatch->swatchHexColor;
108:         }
109:         return $hex;
110:     }
