40:     /**
41:      * Enables creation of a Kuler_Comment object out of a comment XML Item.
42:      * Checks the existance of the 'author' field to make sure the item was
43:      * loaded correctly.
44:      *
45:      * @param SimpleXMLElement $xmlItem
46:      */
47:     public function __construct(SimpleXMLElement $xmlItem)
48:     {
49:         $namespaces = $xmlItem->getDocNamespaces();
50:         $this->_comment = $xmlItem->children($namespaces['kuler'])->commentItem;
51:         if (!isset($this->_comment->author))
52:             throw new Kuler_Exception('Error loading the Comment item');
53:     }
54: 
55:     /**
56:      * Automatic accessors for the $_comment object. The return value is a
57:      * SimpleXMLElement and should be properly casted in most situations.
58:      *
59:      * @param string $name
60:      * @return SimpleXMLElement The value of retrieving $name from the internal $_comment object.
61:      */
62:     public function __get($name)
63:     {
64:         $property = lcfirst($name);
65:         if (property_exists($this->_comment, $property)) {
66:             return $this->_comment->$property;
67:         } else
68:             throw new Kuler_Exception('Property \'' . $name . '\' not found');
69:     }
70: 
71:     /**
72:      * Comments also have some information about the theme they belong to. This
73:      * function creates a Kuler_Theme with that information. <br/><br/>
74:      * Its important to note that doing it this way won't yield the same results
75:      * as loading the theme through the Get RSS (the latter method will return
76:      * a more complete set of properties). In any case, the object's utility
77:      * methods should work in both cases.
78:      *
79:      * @return Kuler_Theme A Kuler_Theme object initialized with the comment's
80:      * theme data.
81:      */
82:     public function getTheme()
83:     {
84:         return new Kuler_Theme($this->_comment);
